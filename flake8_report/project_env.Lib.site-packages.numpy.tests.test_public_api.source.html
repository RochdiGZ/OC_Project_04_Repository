19 def check_dir(module, module_name=None):
20 """Returns a mapping of all objects with the wrong __module__ attribute."""
21 if module_name is None:
22 module_name = module.__name__
23 results = {}
24 for name in dir(module):
25 item = getattr(module, name)
26 if (hasattr(item, '__module__') and hasattr(item, '__name__')
27 and item.__module__ != module_name):
28 results[name] = item.__module__ + '.' + item.__name__
29 return results
