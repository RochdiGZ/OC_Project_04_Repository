17 def get_io_paths(fname_inp, mname="untitled"):
18 """Takes in a temporary file for testing and returns the expected output and input paths
19  
20 Here expected output is essentially one of any of the possible generated
21 files.
22  
23 ..note::
24  
25 Since this does not actually run f2py, none of these are guaranteed to
26 exist, and module names are typically incorrect
27  
28 Parameters
29 ----------
30 fname_inp : str
31 The input filename
32 mname : str, optional
33 The name of the module, untitled by default
34  
35 Returns
36 -------
37 genp : NamedTuple PPaths
38 The possible paths which are generated, not all of which exist
39 """
40 bpath = Path(fname_inp)
41 return PPaths(
42 finp=bpath.with_suffix(".f"),
43 f90inp=bpath.with_suffix(".f90"),
44 pyf=bpath.with_suffix(".pyf"),
45 wrap77=bpath.with_name(f"{mname}-f2pywrappers.f"),
46 wrap90=bpath.with_name(f"{mname}-f2pywrappers2.f90"),
47 cmodf=bpath.with_name(f"{mname}module.c"),
48 )
