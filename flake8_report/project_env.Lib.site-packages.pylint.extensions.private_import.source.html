59 def visit_importfrom(self, node: nodes.ImportFrom) -> None:
60 if utils.is_node_in_typing_guarded_import_block(node):
61 return

63 if self.same_root_dir(node, node.modname):
64 return

68 # Check the imported objects first. If they are all valid type annotations, the package can be private

99 def _get_private_imports(self, names: list[str]) -> list[str]:
100 """Returns the private names from input names by a simple string check."""
101 return [name for name in names if self._name_is_private(name)]

103 @staticmethod
104 def _name_is_private(name: str) -> bool:
105 """Returns true if the name exists, starts with `_`, and if len(name) > 4
106 it is not a dunder, i.e. it does not begin and end with two underscores.
107 """
108 return (
109 bool(name)
110 and name[0] == "_"
111 and (len(name) <= 4 or name[1] != "_" or name[-2:] != "__")
112 )

114 def _get_type_annotation_names(
115 self, node: nodes.Import | nodes.ImportFrom, names: list[str]
116 ) -> list[str]:
117 """Removes from names any names that are used as type annotations with no other
118 illegal usages.
119 """
120 if names and not self.populated_annotations:
121 self._populate_type_annotations(node.root(), self.all_used_type_annotations)
122 self.populated_annotations = True

124 return [
125 n
126 for n in names
127 if n not in self.all_used_type_annotations
128 or (
129 n in self.all_used_type_annotations
130 and not self.all_used_type_annotations[n]

134 def _populate_type_annotations(
135 self, node: nodes.LocalsDictNodeNG, all_used_type_annotations: dict[str, bool]
136 ) -> None:
137 """Adds to `all_used_type_annotations` all names ever used as a type annotation
138 in the node's (nested) scopes and whether they are only used as annotation.
139 """
140 for name in node.locals:
141 # If we find a private type annotation, make sure we do not mask illegal usages
142 private_name = None
143 # All the assignments using this variable that we might have to check for illegal usages later
144 name_assignments = []
145 for usage_node in node.locals[name]:
146 if isinstance(usage_node, nodes.AssignName) and isinstance(
147 usage_node.parent, (nodes.AnnAssign, nodes.Assign)
148 ):
149 assign_parent = usage_node.parent
150 if isinstance(assign_parent, nodes.AnnAssign):
151 name_assignments.append(assign_parent)
152 private_name = self._populate_type_annotations_annotation(
153 usage_node.parent.annotation, all_used_type_annotations
154 )
155 elif isinstance(assign_parent, nodes.Assign):
156 name_assignments.append(assign_parent)
157  
158 if isinstance(usage_node, nodes.FunctionDef):
159 self._populate_type_annotations_function(
160 usage_node, all_used_type_annotations
161 )
162 if isinstance(usage_node, nodes.LocalsDictNodeNG):
163 self._populate_type_annotations(
164 usage_node, all_used_type_annotations
165 )
166 if private_name is not None:
167 # Found a new private annotation, make sure we are not accessing it elsewhere
168 all_used_type_annotations[
169 private_name
170 ] = self._assignments_call_private_name(name_assignments, private_name)

172 def _populate_type_annotations_function(
173 self, node: nodes.FunctionDef, all_used_type_annotations: dict[str, bool]
174 ) -> None:
175 """Adds all names used as type annotation in the arguments and return type of
176 the function node into the dict `all_used_type_annotations`.
177 """
178 if node.args and node.args.annotations:
179 for annotation in node.args.annotations:
180 self._populate_type_annotations_annotation(
181 annotation, all_used_type_annotations
182 )
183 if node.returns:
184 self._populate_type_annotations_annotation(
185 node.returns, all_used_type_annotations

188 def _populate_type_annotations_annotation(
189 self,
190 node: nodes.Attribute | nodes.Subscript | nodes.Name | None,
191 all_used_type_annotations: dict[str, bool],
192 ) -> str | None:
193 """Handles the possibility of an annotation either being a Name, i.e. just type,
194 or a Subscript e.g. `Optional[type]` or an Attribute, e.g. `pylint.lint.linter`.
195 """
196 if isinstance(node, nodes.Name) and node.name not in all_used_type_annotations:
197 all_used_type_annotations[node.name] = True
198 return node.name
199 if isinstance(node, nodes.Subscript): # e.g. Optional[List[str]]
200 # slice is the next nested type
201 self._populate_type_annotations_annotation(
202 node.slice, all_used_type_annotations
203 )
204 # value is the current type name: could be a Name or Attribute
205 return self._populate_type_annotations_annotation(
206 node.value, all_used_type_annotations
207 )

215 @staticmethod
216 def _assignments_call_private_name(
217 assignments: list[nodes.AnnAssign | nodes.Assign], private_name: str
218 ) -> bool:
219 """Returns True if no assignments involve accessing `private_name`."""
220 if all(not assignment.value for assignment in assignments):
221 # Variable annotated but unassigned is not allowed because there may be
222 # possible illegal access elsewhere
223 return False
224 for assignment in assignments:
225 current_attribute = None
226 if isinstance(assignment.value, nodes.Call):
227 current_attribute = assignment.value.func
228 elif isinstance(assignment.value, nodes.Attribute):
229 current_attribute = assignment.value
230 elif isinstance(assignment.value, nodes.Name):
231 current_attribute = assignment.value.name
232 if not current_attribute:
233 continue
234 while isinstance(current_attribute, (nodes.Attribute, nodes.Call)):
235 if isinstance(current_attribute, nodes.Call):
236 current_attribute = current_attribute.func
237 if not isinstance(current_attribute, nodes.Name):
238 current_attribute = current_attribute.expr
239 if (
240 isinstance(current_attribute, nodes.Name)
241 and current_attribute.name == private_name
242 ):
243 return False
244 return True

246 @staticmethod
247 def same_root_dir(
248 node: nodes.Import | nodes.ImportFrom, import_mod_name: str
249 ) -> bool:
250 """Does the node's file's path contain the base name of `import_mod_name`?"""
251 if not import_mod_name: # from . import ...
252 return True
253 if node.level: # from .foo import ..., from ..bar import ...
254 return True
255  
256 base_import_package = import_mod_name.split(".")[0]
257  
258 return base_import_package in Path(node.root().file).parent.parts
