59 class Super(node_classes.NodeNG):
60 """Proxy class over a super call.
61  
62 This class offers almost the same behaviour as Python's super,
63 which is MRO lookups for retrieving attributes from the parents.
64  
65 The *mro_pointer* is the place in the MRO from where we should
66 start looking, not counting it. *mro_type* is the object which
67 provides the MRO, it can be both a type or an instance.
68 *self_class* is the class where the super call is, while
69 *scope* is the function where the super call is.
70 """

141 def igetattr(self, name: str, context: InferenceContext | None = None):
142 """Retrieve the inferred values of the given attribute name."""
143 # '__class__' is a special attribute that should be taken directly
144 # from the special attributes dict
145 if name == "__class__":
146 yield self.special_attributes.lookup(name)
147 return
148  
149 try:
150 mro = self.super_mro()
151 # Don't let invalid MROs or invalid super calls
152 # leak out as is from this function.
153 except SuperError as exc:
154 raise AttributeInferenceError(
155 (
156 "Lookup for {name} on {target!r} because super call {super!r} "
157 "is invalid."
158 ),
159 target=self,
160 attribute=name,
161 context=context,
162 super_=exc.super_,
163 ) from exc
164 except MroError as exc:
165 raise AttributeInferenceError(
166 (
167 "Lookup for {name} on {target!r} failed because {cls!r} has an "
168 "invalid MRO."
169 ),
170 target=self,
171 attribute=name,
172 context=context,
173 mros=exc.mros,
174 cls=exc.cls,
175 ) from exc
176 found = False
177 for cls in mro:
178 if name not in cls.locals:
179 continue
180  
181 found = True
182 for inferred in bases._infer_stmts([cls[name]], context, frame=self):
183 if not isinstance(inferred, scoped_nodes.FunctionDef):
184 yield inferred
185 continue
186  
187 # We can obtain different descriptors from a super depending
188 # on what we are accessing and where the super call is.
189 if inferred.type == "classmethod":
190 yield bases.BoundMethod(inferred, cls)
191 elif self._scope.type == "classmethod" and inferred.type == "method":
192 yield inferred
193 elif self._class_based or inferred.type == "staticmethod":
194 yield inferred
195 elif isinstance(inferred, Property):
196 function = inferred.function
197 try:
198 yield from function.infer_call_result(
199 caller=self, context=context
200 )
201 except InferenceError:
202 yield util.Uninferable
203 elif bases._is_property(inferred):
204 # TODO: support other descriptors as well.
205 try:
206 yield from inferred.infer_call_result(self, context)
207 except InferenceError:
208 yield util.Uninferable
209 else:
210 yield bases.BoundMethod(inferred, cls)
211  
212 # Only if we haven't found any explicit overwrites for the
213 # attribute we look it up in the special attributes
214 if not found and name in self.special_attributes:
215 yield self.special_attributes.lookup(name)
216 return
217  
218 if not found:
219 raise AttributeInferenceError(target=self, attribute=name, context=context)
220  
221 def getattr(self, name, context=None):
222 return list(self.igetattr(name, context=context))

271 class PartialFunction(scoped_nodes.FunctionDef):
272 """A class representing partial function obtained via functools.partial"""
273  
274 @decorators.deprecate_arguments(doc="Use the postinit arg 'doc_node' instead")
275 def __init__(
276 self, call, name=None, doc=None, lineno=None, col_offset=None, parent=None
277 ):
278 # TODO: Pass end_lineno and end_col_offset as well
279 super().__init__(name, lineno=lineno, col_offset=col_offset, parent=None)
280 # Assigned directly to prevent triggering the DeprecationWarning.
281 self._doc = doc
282 # A typical FunctionDef automatically adds its name to the parent scope,
283 # but a partial should not, so defer setting parent until after init
284 self.parent = parent
285 self.filled_args = call.positional_arguments[1:]
286 self.filled_keywords = call.keyword_arguments
287  
288 wrapped_function = call.positional_arguments[0]
289 inferred_wrapped_function = next(wrapped_function.infer())
290 if isinstance(inferred_wrapped_function, PartialFunction):
291 self.filled_args = inferred_wrapped_function.filled_args + self.filled_args
292 self.filled_keywords = {
293 **inferred_wrapped_function.filled_keywords,
294 **self.filled_keywords,
295 }
296  
297 self.filled_positionals = len(self.filled_args)
298  
299 def infer_call_result(self, caller=None, context=None):
300 if context:
301 current_passed_keywords = {
302 keyword for (keyword, _) in context.callcontext.keywords
303 }
304 for keyword, value in self.filled_keywords.items():
305 if keyword not in current_passed_keywords:
306 context.callcontext.keywords.append((keyword, value))
307  
308 call_context_args = context.callcontext.args or []
309 context.callcontext.args = self.filled_args + call_context_args
310  
311 return super().infer_call_result(caller=caller, context=context)
