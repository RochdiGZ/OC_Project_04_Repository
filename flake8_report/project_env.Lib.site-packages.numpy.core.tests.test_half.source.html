89 # Assumes that round to even is used during casting.
90 max_pattern = np.float16(np.finfo(np.float16).max).view(np.uint16)
91  
92 # Test all (positive) finite numbers, denormals are most interesting
93 # however:
94 f16s_patterns = np.arange(0, max_pattern+1, dtype=np.uint16)
95 f16s_float = f16s_patterns.view(np.float16).astype(float_t)
96  
97 # Shift the values by half a bit up or a down (or do not shift),
98 if shift == "up":
99 f16s_float = 0.5 * (f16s_float[:-1] + f16s_float[1:])[1:]
100 elif shift == "down":
101 f16s_float = 0.5 * (f16s_float[:-1] + f16s_float[1:])[:-1]
102 else:
103 f16s_float = f16s_float[1:-1]
104  
105 # Increase the float by a minimal value:
106 if offset == "up":
107 f16s_float = np.nextafter(f16s_float, float_t(1e50))
108 elif offset == "down":
109 f16s_float = np.nextafter(f16s_float, float_t(-1e50))
110  
111 # Convert back to float16 and its bit pattern:
112 res_patterns = f16s_float.astype(np.float16).view(np.uint16)
113  
114 # The above calculations tries the original values, or the exact
115 # mid points between the float16 values. It then further offsets them
116 # by as little as possible. If no offset occurs, "round to even"
117 # logic will be necessary, an arbitrarily small offset should cause
118 # normal up/down rounding always.
119  
120 # Calculate the expected pattern:
121 cmp_patterns = f16s_patterns[1:-1].copy()
122  
123 if shift == "down" and offset != "up":
124 shift_pattern = -1
125 elif shift == "up" and offset != "down":
126 shift_pattern = 1
127 else:
128 # There cannot be a shift, either shift is None, so all rounding
129 # will go back to original, or shift is reduced by offset too much.
130 shift_pattern = 0
131  
132 # If rounding occurs, is it normal rounding or round to even?
133 if offset is None:
134 # Round to even occurs, modify only non-even, cast to allow + (-1)
135 cmp_patterns[0::2].view(np.int16)[...] += shift_pattern
136 else:
137 cmp_patterns.view(np.int16)[...] += shift_pattern
138  
139 assert_equal(res_patterns, cmp_patterns)
