27 def findcommonblocks(block, top=1):
28 ret = []
29 if hascommon(block):
30 for key, value in block['common'].items():
31 vars_ = {v: block['vars'][v] for v in value}
32 ret.append((key, value, vars_))
33 elif hasbody(block):
34 for b in block['body']:
35 ret = ret + findcommonblocks(b, 0)
36 if top:
37 tret = []
38 names = []
39 for t in ret:
40 if t[0] not in names:
41 names.append(t[0])
42 tret.append(t)
43 return tret
44 return ret

47 def buildhooks(m):
48 ret = {'commonhooks': [], 'initcommonhooks': [],
49 'docs': ['"COMMON blocks:\\n"']}
50 fwrap = ['']
51  
52 def fadd(line, s=fwrap):
53 s[0] = '%s\n %s' % (s[0], line)
54 chooks = ['']

56 def cadd(line, s=chooks):
57 s[0] = '%s\n%s' % (s[0], line)
58 ihooks = ['']
59  
60 def iadd(line, s=ihooks):
61 s[0] = '%s\n%s' % (s[0], line)
62 doc = ['']
63  
64 def dadd(line, s=doc):
65 s[0] = '%s\n%s' % (s[0], line)
66 for (name, vnames, vars) in findcommonblocks(m):
67 lower_name = name.lower()
68 hnames, inames = [], []
69 for n in vnames:
70 if isintent_hide(vars[n]):
71 hnames.append(n)
72 else:
73 inames.append(n)
74 if hnames:
75 outmess('\t\tConstructing COMMON block support for "%s"...\n\t\t %s\n\t\t Hidden: %s\n' % (
76 name, ','.join(inames), ','.join(hnames)))
77 else:
78 outmess('\t\tConstructing COMMON block support for "%s"...\n\t\t %s\n' % (
79 name, ','.join(inames)))
80 fadd('subroutine f2pyinit%s(setupfunc)' % name)
81 fadd('external setupfunc')
82 for n in vnames:
83 fadd(func2subr.var2fixfortran(vars, n))
84 if name == '_BLNK_':
85 fadd('common %s' % (','.join(vnames)))
86 else:
87 fadd('common /%s/ %s' % (name, ','.join(vnames)))
88 fadd('call setupfunc(%s)' % (','.join(inames)))
89 fadd('end\n')
90 cadd('static FortranDataDef f2py_%s_def[] = {' % (name))
91 idims = []
92 for n in inames:
93 ct = capi_maps.getctype(vars[n])
94 at = capi_maps.c2capi_map[ct]
95 dm = capi_maps.getarrdims(n, vars[n])
96 if dm['dims']:
97 idims.append('(%s)' % (dm['dims']))
98 else:
99 idims.append('')
100 dms = dm['dims'].strip()
101 if not dms:
102 dms = '-1'
103 cadd('\t{\"%s\",%s,{{%s}},%s},' % (n, dm['rank'], dms, at))
104 cadd('\t{NULL}\n};')
105 inames1 = rmbadname(inames)
106 inames1_tps = ','.join(['char *' + s for s in inames1])
107 cadd('static void f2py_setup_%s(%s) {' % (name, inames1_tps))
108 cadd('\tint i_f2py=0;')
109 for n in inames1:
110 cadd('\tf2py_%s_def[i_f2py++].data = %s;' % (name, n))
111 cadd('}')
112 if '_' in lower_name:
113 F_FUNC = 'F_FUNC_US'
114 else:
115 F_FUNC = 'F_FUNC'
116 cadd('extern void %s(f2pyinit%s,F2PYINIT%s)(void(*)(%s));'
117 % (F_FUNC, lower_name, name.upper(),
118 ','.join(['char*'] * len(inames1))))
119 cadd('static void f2py_init_%s(void) {' % name)
120 cadd('\t%s(f2pyinit%s,F2PYINIT%s)(f2py_setup_%s);'
121 % (F_FUNC, lower_name, name.upper(), name))
122 cadd('}\n')
123 iadd('\ttmp = PyFortranObject_New(f2py_%s_def,f2py_init_%s);' % (name, name))
124 iadd('\tF2PyDict_SetItemString(d, \"%s\", tmp);' % name)
125 iadd('\tPy_DECREF(tmp);')
126 tname = name.replace('_', '\\_')
127 dadd('\\subsection{Common block \\texttt{%s}}\n' % (tname))
128 dadd('\\begin{description}')
129 for n in inames:
130 dadd('\\item[]{{}\\verb@%s@{}}' %
131 (capi_maps.getarrdocsign(n, vars[n])))
132 if hasnote(vars[n]):
133 note = vars[n]['note']
134 if isinstance(note, list):
135 note = '\n'.join(note)
136 dadd('--- %s' % (note))
137 dadd('\\end{description}')
138 ret['docs'].append(
139 '"\t/%s/ %s\\n"' % (name, ','.join(map(lambda v, d: v + d, inames, idims))))
140 ret['commonhooks'] = chooks
141 ret['initcommonhooks'] = ihooks
142 ret['latexdoc'] = doc[0]
143 if len(ret['docs']) <= 1:
144 ret['docs'] = ''
145 return ret, fwrap[0]
