109 def is_method_call(
110 func: bases.BoundMethod, types: tuple[str, ...] = (), methods: tuple[str, ...] = ()
111 ) -> bool:
112 """Determines if a BoundMethod node represents a method call.
113  
114 Args:
115 func: The BoundMethod AST node to check.
116 types: Optional sequence of caller type names to restrict check.
117 methods: Optional sequence of method names to restrict check.
118  
119 Returns:
120 true if the node represents a method call for the given type and
121 method names, False otherwise.
122 """
123 return (
124 isinstance(func, astroid.BoundMethod)
125 and isinstance(func.bound, astroid.Instance)
126 and (func.bound.name in types if types else True)
127 and (func.name in methods if methods else True)
128 )

161 def visit_module(self, _: nodes.Module) -> None:
162 """Clears any state left in this checker from last module checked."""
163 # The code being checked can just as easily "import logging as foo",
164 # so it is necessary to process the imports and store in this field
165 # what name the logging module is actually given.
166 self._logging_names: set[str] = set()
167 logging_mods = self.linter.config.logging_modules
168  
169 self._format_style = self.linter.config.logging_format_style
170  
171 self._logging_modules = set(logging_mods)
172 self._from_imports = {}
173 for logging_mod in logging_mods:
174 parts = logging_mod.rsplit(".", 1)
175 if len(parts) > 1:
176 self._from_imports[parts[0]] = parts[1]

178 def visit_importfrom(self, node: nodes.ImportFrom) -> None:
179 """Checks to see if a module uses a non-Python logging module."""
180 try:
181 logging_name = self._from_imports[node.modname]
182 for module, as_name in node.names:
183 if module == logging_name:
184 self._logging_names.add(as_name or module)
185 except KeyError:
186 pass

189 """Checks to see if this module uses Python's built-in logging."""

204 def is_logger_class() -> tuple[bool, str | None]:
205 for inferred in infer_all(node.func):
206 if isinstance(inferred, astroid.BoundMethod):
207 parent = inferred._proxied.parent
208 if isinstance(parent, nodes.ClassDef) and (
209 parent.qname() == "logging.Logger"
210 or any(
211 ancestor.qname() == "logging.Logger"
212 for ancestor in parent.ancestors()
213 )
214 ):
215 return True, inferred._proxied.name
216 return False, None

219 name = node.func.attrname
220 else:
221 result, name = is_logger_class()

226 def _check_log_method(self, node: nodes.Call, name: str) -> None:
227 """Checks calls to logging.log(level, format, *format_args)."""
228 if name == "log":
229 if node.starargs or node.kwargs or len(node.args) < 2:
230 # Either a malformed call, star args, or double-star args. Beyond
231 # the scope of this checker.
232 return
233 format_pos: Literal[0, 1] = 1
234 elif name in CHECKED_CONVENIENCE_FUNCTIONS:
235 if node.starargs or node.kwargs or not node.args:
236 # Either no args, star args, or double-star args. Beyond the
237 # scope of this checker.
238 return
239 format_pos = 0
240 else:
241 return
242  
243 if isinstance(node.args[format_pos], nodes.BinOp):
244 binop = node.args[format_pos]
245 emit = binop.op == "%"
246 if binop.op == "+":
247 total_number_of_strings = sum(
248 1
249 for operand in (binop.left, binop.right)
250 if self._is_operand_literal_str(utils.safe_infer(operand))
251 )
252 emit = total_number_of_strings > 0
253 if emit:
254 self.add_message(
255 "logging-not-lazy",
256 node=node,
257 args=(self._helper_string(node),),
258 )
259 elif isinstance(node.args[format_pos], nodes.Call):
260 self._check_call_func(node.args[format_pos])
261 elif isinstance(node.args[format_pos], nodes.Const):
262 self._check_format_string(node, format_pos)
263 elif isinstance(node.args[format_pos], nodes.JoinedStr):
264 self.add_message(
265 "logging-fstring-interpolation",
266 node=node,
267 args=(self._helper_string(node),),
268 )

270 def _helper_string(self, node: nodes.Call) -> str:
271 """Create a string that lists the valid types of formatting for this node."""
272 valid_types = ["lazy %"]
273  
274 if not self.linter.is_message_enabled(
275 "logging-fstring-formatting", node.fromlineno
276 ):
277 valid_types.append("fstring")
278 if not self.linter.is_message_enabled(
279 "logging-format-interpolation", node.fromlineno
280 ):
281 valid_types.append(".format()")
282 if not self.linter.is_message_enabled("logging-not-lazy", node.fromlineno):
283 valid_types.append("%")
284  
285 return " or ".join(valid_types)

293 """Checks that function call is not format_string.format()."""

365 def is_complex_format_str(node: nodes.NodeNG) -> bool:
366 """Return whether the node represents a string with complex formatting specs."""
367 inferred = utils.safe_infer(node)
368 if inferred is None or not (
369 isinstance(inferred, nodes.Const) and isinstance(inferred.value, str)
370 ):
371 return True
372 try:
373 parsed = list(string.Formatter().parse(inferred.value))
374 except ValueError:
375 # This format string is invalid
376 return False
377 return any(format_spec for (_, _, format_spec, _) in parsed)

380 def _count_supplied_tokens(args: list[nodes.NodeNG]) -> int:
381 """Counts the number of tokens in an args list.
382  
383 The Python log functions allow for special keyword arguments: func,
384 exc_info and extra. To handle these cases correctly, we only count
385 arguments that aren't keywords.
386  
387 Args:
388 args: AST nodes that are arguments for a log format string.
389  
390 Returns:
391 Number of AST nodes that aren't keywords.
392 """
393 return sum(1 for arg in args if not isinstance(arg, nodes.Keyword))
