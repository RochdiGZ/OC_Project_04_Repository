125 def _last_token_on_line_is(tokens: TokenWrapper, line_end: int, token: str) -> bool:
126 return (
127 line_end > 0
128 and tokens.token(line_end - 1) == token

131 and tokens.type(line_end - 1) == tokenize.COMMENT

282 def _check_keyword_parentheses(
283 self, tokens: list[tokenize.TokenInfo], start: int
284 ) -> None:
285 """Check that there are not unnecessary parentheses after a keyword.
286  
287 Parens are unnecessary if there is exactly one balanced outer pair on a
288 line and contains no commas (i.e. is not a tuple).
289  
290 Args:
291 tokens: The entire list of Tokens.
292 start: The position of the keyword in the token list.
293 """
294 # If the next token is not a paren, we're fine.
295 if tokens[start + 1].string != "(":
296 return
297 if (
298 tokens[start].string == "not"
299 and start > 0
300 and tokens[start - 1].string == "is"
301 ):
302 # If this is part of an `is not` expression, we have a binary operator
303 # so the parentheses are not necessarily redundant.
304 return
305 found_and_or = False
306 contains_walrus_operator = False
307 walrus_operator_depth = 0
308 contains_double_parens = 0
309 depth = 0
310 keyword_token = str(tokens[start].string)
311 line_num = tokens[start].start[0]
312 for i in range(start, len(tokens) - 1):
313 token = tokens[i]
314  
315 # If we hit a newline, then assume any parens were for continuation.
316 if token.type == tokenize.NL:
317 return
318 # Since the walrus operator doesn't exist below python3.8, the tokenizer
319 # generates independent tokens
320 if (
321 token.string == ":=" # <-- python3.8+ path
322 or token.string + tokens[i + 1].string == ":="
323 ):
324 contains_walrus_operator = True
325 walrus_operator_depth = depth
326 if token.string == "(":
327 depth += 1
328 if tokens[i + 1].string == "(":
329 contains_double_parens = 1
330 elif token.string == ")":
331 depth -= 1
332 if depth:
333 if contains_double_parens and tokens[i + 1].string == ")":
334 # For walrus operators in `if (not)` conditions and comprehensions
335 if keyword_token in {"in", "if", "not"}:
336 continue
337 return
338 contains_double_parens -= 1
339 continue
340 # ')' can't happen after if (foo), since it would be a syntax error.
341 if tokens[i + 1].string in {":", ")", "]", "}", "in"} or tokens[
342 i + 1
343 ].type in {tokenize.NEWLINE, tokenize.ENDMARKER, tokenize.COMMENT}:
344 if contains_walrus_operator and walrus_operator_depth - 1 == depth:
345 return
346 # The empty tuple () is always accepted.
347 if i == start + 2:
348 return
349 if found_and_or:
350 return
351 if keyword_token == "in":
352 # This special case was added in https://github.com/PyCQA/pylint/pull/4948
353 # but it could be removed in the future. Avoid churn for now.
354 return
355 self.add_message(
356 "superfluous-parens", line=line_num, args=keyword_token
357 )
358 return
359 elif depth == 1:
360 # This is a tuple, which is always acceptable.
361 if token[1] == ",":
362 return
363 # 'and' and 'or' are the only boolean operators with lower precedence
364 # than 'not', so parens are only required when they are found.
365 if token[1] in {"and", "or"}:
366 found_and_or = True
367 # A yield inside an expression must always be in parentheses,
368 # quit early without error.
369 elif token[1] == "yield":
370 return
371 # A generator expression always has a 'for' token in it, and
372 # the 'for' token is only legal inside parens when it is in a
373 # generator expression. The parens are necessary here, so bail
374 # without an error.
375 elif token[1] == "for":
376 return
377 # A generator expression can have an 'else' token in it.
378 # We check the rest of the tokens to see if any problems occur after
379 # the 'else'.
380 elif token[1] == "else":
381 if "(" in (i.string for i in tokens[i:]):
382 self._check_keyword_parentheses(tokens[i:], 0)
383 return

385 def process_tokens(self, tokens: list[tokenize.TokenInfo]) -> None:
386 """Process tokens and search for:
387  
388 - too long lines (i.e. longer than <max_chars>)
389 - optionally bad construct (if given, bad_construct must be a compiled
390 regular expression).
391 """
392 indents = [0]
393 check_equal = False
394 line_num = 0
395 self._lines = {}
396 self._visited_lines = {}
397 self._last_line_ending: str | None = None
398 last_blank_line_num = 0
399 for idx, (tok_type, string, start, _, line) in enumerate(tokens):
400 if start[0] != line_num:
401 line_num = start[0]
402 # A tokenizer oddity: if an indented line contains a multi-line
403 # docstring, the line member of the INDENT token does not contain
404 # the full line; therefore we check the next token on the line.
405 if tok_type == tokenize.INDENT:
406 self.new_line(TokenWrapper(tokens), idx - 1, idx + 1)
407 # A tokenizer oddity: if a line contains a multi-line string,
408 # the NEWLINE also gets its own token which we already checked in
409 # the multi-line docstring case.
410 # See https://github.com/PyCQA/pylint/issues/6936
411 elif tok_type == tokenize.NEWLINE:
412 pass
413 else:
414 self.new_line(TokenWrapper(tokens), idx - 1, idx)
415  
416 if tok_type == tokenize.NEWLINE:
417 # a program statement, or ENDMARKER, will eventually follow,
418 # after some (possibly empty) run of tokens of the form
419 # (NL | COMMENT)* (INDENT | DEDENT+)?
420 # If an INDENT appears, setting check_equal is wrong, and will
421 # be undone when we see the INDENT.
422 check_equal = True
423 self._check_line_ending(string, line_num)
424 elif tok_type == tokenize.INDENT:
425 check_equal = False
426 self.check_indent_level(string, indents[-1] + 1, line_num)
427 indents.append(indents[-1] + 1)
428 elif tok_type == tokenize.DEDENT:
429 # there's nothing we need to check here! what's important is
430 # that when the run of DEDENTs ends, the indentation of the
431 # program statement (or ENDMARKER) that triggered the run is
432 # equal to what's left at the top of the indents stack
433 check_equal = True
434 if len(indents) > 1:
435 del indents[-1]
436 elif tok_type == tokenize.NL:
437 if not line.strip("\r\n"):
438 last_blank_line_num = line_num
439 elif tok_type not in (tokenize.COMMENT, tokenize.ENCODING):
440 # This is the first concrete token following a NEWLINE, so it
441 # must be the first token of the next program statement, or an
442 # ENDMARKER; the "line" argument exposes the leading white-space
443 # for this statement; in the case of ENDMARKER, line is an empty
444 # string, so will properly match the empty string with which the
445 # "indents" stack was seeded
446 if check_equal:
447 check_equal = False
448 self.check_indent_level(line, indents[-1], line_num)
449  
450 if tok_type == tokenize.NUMBER and string.endswith("l"):
451 self.add_message("lowercase-l-suffix", line=line_num)
452  
453 if string in _KEYWORD_TOKENS:
454 self._check_keyword_parentheses(tokens, idx)
455  
456 line_num -= 1 # to be ok with "wc -l"
457 if line_num > self.linter.config.max_module_lines:
458 # Get the line where the too-many-lines (or its message id)
459 # was disabled or default to 1.
460 message_definition = self.linter.msgs_store.get_message_definitions(
461 "too-many-lines"
462 )[0]
463 names = (message_definition.msgid, "too-many-lines")
464 lineno = next(
465 filter(None, (self.linter._pragma_lineno.get(name) for name in names)),
466 1,
467 )
468 self.add_message(
469 "too-many-lines",
470 args=(line_num, self.linter.config.max_module_lines),
471 line=lineno,
472 )
473  
474 # See if there are any trailing lines. Do not complain about empty
475 # files like __init__.py markers.
476 if line_num == last_blank_line_num and line_num > 0:
477 self.add_message("trailing-newlines", line=line_num)

479 def _check_line_ending(self, line_ending: str, line_num: int) -> None:
480 # check if line endings are mixed
481 if self._last_line_ending is not None:
482 # line_ending == "" indicates a synthetic newline added at
483 # the end of a file that does not, in fact, end with a
484 # newline.
485 if line_ending and line_ending != self._last_line_ending:
486 self.add_message("mixed-line-endings", line=line_num)
487  
488 self._last_line_ending = line_ending
489  
490 # check if line ending is as expected
491 expected = self.linter.config.expected_line_ending_format
492 if expected:
493 # reduce multiple \n\n\n\n to one \n
494 line_ending = reduce(lambda x, y: x + y if x != y else x, line_ending, "")
495 line_ending = "LF" if line_ending == "\n" else "CRLF"
496 if line_ending != expected:
497 self.add_message(
498 "unexpected-line-ending-format",
499 args=(line_ending, expected),
500 line=line_num,
501 )

503 @only_required_for_messages("multiple-statements")
504 def visit_default(self, node: nodes.NodeNG) -> None:
505 """Check the node line number and check it if not yet done."""
506 if not node.is_statement:
507 return
508 if not node.root().pure_python:
509 return
510 prev_sibl = node.previous_sibling()
511 if prev_sibl is not None:
512 prev_line = prev_sibl.fromlineno
513 # The line on which a 'finally': occurs in a 'try/finally'
514 # is not directly represented in the AST. We infer it
515 # by taking the last line of the body and adding 1, which
516 # should be the line of finally:
517 elif (
518 isinstance(node.parent, nodes.TryFinally) and node in node.parent.finalbody
519 ):
520 prev_line = node.parent.body[0].tolineno + 1
521 elif isinstance(node.parent, nodes.Module):
522 prev_line = 0
523 else:
524 prev_line = node.parent.statement(future=True).fromlineno
525 line = node.fromlineno
526 assert line, node
527 if prev_line == line and self._visited_lines.get(line) != 2:
528 self._check_multi_statement_line(node, line)
529 return
530 if line in self._visited_lines:
531 return
532 try:
533 tolineno = node.blockstart_tolineno
534 except AttributeError:
535 tolineno = node.tolineno
536 assert tolineno, node
537 lines: list[str] = []
538 for line in range(line, tolineno + 1):
539 self._visited_lines[line] = 1
540 try:
541 lines.append(self._lines[line].rstrip())
542 except KeyError:
543 lines.append("")

545 def _check_multi_statement_line(self, node: nodes.NodeNG, line: int) -> None:
546 """Check for lines containing multiple statements."""
547 # Do not warn about multiple nested context managers
548 # in with statements.
549 if isinstance(node, nodes.With):
550 return
551 # For try... except... finally..., the two nodes
552 # appear to be on the same line due to how the AST is built.
553 if isinstance(node, nodes.TryExcept) and isinstance(
554 node.parent, nodes.TryFinally
555 ):
556 return

558 isinstance(node.parent, nodes.If)
559 and not node.parent.orelse
560 and self.linter.config.single_line_if_stmt
561 ):
562 return
563 if (
564 isinstance(node.parent, nodes.ClassDef)
565 and len(node.parent.body) == 1
566 and self.linter.config.single_line_class_stmt

568 return

572 isinstance(node, nodes.Expr)

633 @staticmethod
634 def specific_splitlines(lines: str) -> list[str]:
635 """Split lines according to universal newlines except those in a specific
636 sets.
637 """
638 unsplit_ends = {
639 "\x0b", # synonym of \v
640 "\x0c", # synonym of \f
641 "\x1c",
642 "\x1d",
643 "\x1e",
644 "\x85",
645 "\u2028",
646 "\u2029",
647 }
648 res: list[str] = []
649 buffer = ""
650 for atomic_line in lines.splitlines(True):
651 if atomic_line[-1] not in unsplit_ends:
652 res.append(buffer + atomic_line)
653 buffer = ""
654 else:
655 buffer += atomic_line
656 return res

658 def check_lines(self, lines: str, lineno: int) -> None:
659 """Check given lines for potential messages.
660  
661 Check if lines have:
662 - a final newline
663 - no trailing white-space
664 - less than a maximum number of characters
665 """
666 # we're first going to do a rough check whether any lines in this set
667 # go over the line limit. If none of them do, then we don't need to
668 # parse out the pylint options later on and can just assume that these
669 # lines are clean
670  
671 # we'll also handle the line ending check here to avoid double-iteration
672 # unless the line lengths are suspect
673  
674 max_chars = self.linter.config.max_line_length
675  
676 split_lines = self.specific_splitlines(lines)
677  
678 for offset, line in enumerate(split_lines):
679 self.check_line_ending(line, lineno + offset)
680  
681 # hold onto the initial lineno for later
682 potential_line_length_warning = False
683 for offset, line in enumerate(split_lines):
684 # this check is purposefully simple and doesn't rstrip
685 # since this is running on every line you're checking it's
686 # advantageous to avoid doing a lot of work
687 if len(line) > max_chars:
688 potential_line_length_warning = True
689 break
690  
691 # if there were no lines passing the max_chars config, we don't bother
692 # running the full line check (as we've met an even more strict condition)
693 if not potential_line_length_warning:
694 return
695  
696 # Line length check may be deactivated through `pylint: disable` comment
697 mobj = OPTION_PO.search(lines)
698 checker_off = False
699 if mobj:
700 if not self.is_line_length_check_activated(mobj):
701 checker_off = True
702 # The 'pylint: disable whatever' should not be taken into account for line length count
703 lines = self.remove_pylint_option_from_lines(mobj)
704  
705 # here we re-run specific_splitlines since we have filtered out pylint options above
706 for offset, line in enumerate(self.specific_splitlines(lines)):
707 self.check_line_length(line, lineno + offset, checker_off)
