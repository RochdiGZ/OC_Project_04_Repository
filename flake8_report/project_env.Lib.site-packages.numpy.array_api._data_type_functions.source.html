47 def can_cast(from_: Union[Dtype, Array], to: Dtype, /) -> bool:
48 """
49 Array API compatible wrapper for :py:func:`np.can_cast <numpy.can_cast>`.
50  
51 See its docstring for more information.
52 """
53 if isinstance(from_, Array):
54 from_ = from_.dtype
55 elif from_ not in _all_dtypes:
56 raise TypeError(f"{from_=}, but should be an array_api array or dtype")
57 if to not in _all_dtypes:
58 raise TypeError(f"{to=}, but should be a dtype")
59 # Note: We avoid np.can_cast() as it has discrepancies with the array API,
60 # since NumPy allows cross-kind casting (e.g., NumPy allows bool -> int8).
61 # See https://github.com/numpy/numpy/issues/20870
62 try:
63 # We promote `from_` and `to` together. We then check if the promoted
64 # dtype is `to`, which indicates if `from_` can (up)cast to `to`.
65 dtype = _result_type(from_, to)
66 return to == dtype
67 except TypeError:
68 # _result_type() raises if the dtypes don't promote together
69 return False

120 def result_type(*arrays_and_dtypes: Union[Array, Dtype]) -> Dtype:
121 """
122 Array API compatible wrapper for :py:func:`np.result_type <numpy.result_type>`.
123  
124 See its docstring for more information.
125 """
126 # Note: we use a custom implementation that gives only the type promotions
127 # required by the spec rather than using np.result_type. NumPy implements
128 # too many extra type promotions like int64 + uint64 -> float64, and does
129 # value-based casting on scalar arrays.
130 A = []
131 for a in arrays_and_dtypes:
132 if isinstance(a, Array):
133 a = a.dtype
134 elif isinstance(a, np.ndarray) or a not in _all_dtypes:
135 raise TypeError("result_type() inputs must be array_api arrays or dtypes")
136 A.append(a)
137  
138 if len(A) == 0:
139 raise ValueError("at least one array or dtype is required")
140 elif len(A) == 1:
141 return A[0]
142 else:
143 t = A[0]
144 for t2 in A[1:]:
145 t = _result_type(t, t2)
146 return t
