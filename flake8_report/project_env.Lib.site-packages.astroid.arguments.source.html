14 class CallSite:
15 """Class for understanding arguments passed into a call site
16  
17 It needs a call context, which contains the arguments and the
18 keyword arguments that were passed into a given call site.
19 In order to infer what an argument represents, call :meth:`infer_argument`
20 with the corresponding function node and the argument name.
21  
22 :param callcontext:
23 An instance of :class:`astroid.context.CallContext`, that holds
24 the arguments for the call site.
25 :param argument_context_map:
26 Additional contexts per node, passed in from :attr:`astroid.context.Context.extra_context`
27 :param context:
28 An instance of :class:`astroid.context.Context`.
29 """
30  
31 def __init__(
32 self, callcontext: CallContext, argument_context_map=None, context=None
33 ):
34 if argument_context_map is None:
35 argument_context_map = {}
36 self.argument_context_map = argument_context_map
37 args = callcontext.args
38 keywords = callcontext.keywords
39 self.duplicated_keywords: set[str] = set()
40 self._unpacked_args = self._unpack_args(args, context=context)
41 self._unpacked_kwargs = self._unpack_keywords(keywords, context=context)
42  
43 self.positional_arguments = [
44 arg for arg in self._unpacked_args if arg is not Uninferable
45 ]
46 self.keyword_arguments = {
47 key: value
48 for key, value in self._unpacked_kwargs.items()
49 if value is not Uninferable
50 }

52 @classmethod
53 def from_call(cls, call_node, context: InferenceContext | None = None):
54 """Get a CallSite object from the given Call node.
55  
56 context will be used to force a single inference path.
57 """
58  
59 # Determine the callcontext from the given `context` object if any.
60 context = context or InferenceContext()
61 callcontext = CallContext(call_node.args, call_node.keywords)
62 return cls(callcontext, context=context)

64 def has_invalid_arguments(self):
65 """Check if in the current CallSite were passed *invalid* arguments
66  
67 This can mean multiple things. For instance, if an unpacking
68 of an invalid object was passed, then this method will return True.
69 Other cases can be when the arguments can't be inferred by astroid,
70 for example, by passing objects which aren't known statically.
71 """
72 return len(self.positional_arguments) != len(self._unpacked_args)
73  
74 def has_invalid_keywords(self):
75 """Check if in the current CallSite were passed *invalid* keyword arguments
76  
77 For instance, unpacking a dictionary with integer keys is invalid
78 (**{1:2}), because the keys must be strings, which will make this
79 method to return True. Other cases where this might return True if
80 objects which can't be inferred were passed.
81 """
82 return len(self.keyword_arguments) != len(self._unpacked_kwargs)

84 def _unpack_keywords(self, keywords, context=None):
85 values = {}
86 context = context or InferenceContext()
87 context.extra_context = self.argument_context_map
88 for name, value in keywords:
89 if name is None:
90 # Then it's an unpacking operation (**)
91 try:
92 inferred = next(value.infer(context=context))
93 except InferenceError:
94 values[name] = Uninferable
95 continue
96 except StopIteration:
97 continue
98  
99 if not isinstance(inferred, nodes.Dict):
100 # Not something we can work with.
101 values[name] = Uninferable
102 continue
103  
104 for dict_key, dict_value in inferred.items:
105 try:
106 dict_key = next(dict_key.infer(context=context))
107 except InferenceError:
108 values[name] = Uninferable
109 continue
110 except StopIteration:
111 continue
112 if not isinstance(dict_key, nodes.Const):
113 values[name] = Uninferable
114 continue
115 if not isinstance(dict_key.value, str):
116 values[name] = Uninferable
117 continue
118 if dict_key.value in values:
119 # The name is already in the dictionary
120 values[dict_key.value] = Uninferable
121 self.duplicated_keywords.add(dict_key.value)
122 continue
123 values[dict_key.value] = dict_value
124 else:
125 values[name] = value
126 return values

128 def _unpack_args(self, args, context=None):
129 values = []
130 context = context or InferenceContext()
131 context.extra_context = self.argument_context_map
132 for arg in args:
133 if isinstance(arg, nodes.Starred):
134 try:
135 inferred = next(arg.value.infer(context=context))
136 except InferenceError:
137 values.append(Uninferable)
138 continue
139 except StopIteration:
140 continue
141  
142 if inferred is Uninferable:
143 values.append(Uninferable)
144 continue
145 if not hasattr(inferred, "elts"):
146 values.append(Uninferable)
147 continue
148 values.extend(inferred.elts)
149 else:
150 values.append(arg)
151 return values

153 def infer_argument(self, funcnode, name, context):
154 """infer a function argument value according to the call context
155  
156 Arguments:
157 funcnode: The function being called.
158 name: The name of the argument whose value is being inferred.
159 context: Inference context object
160 """
161 if name in self.duplicated_keywords:
162 raise InferenceError(
163 "The arguments passed to {func!r} have duplicate keywords.",
164 call_site=self,
165 func=funcnode,
166 arg=name,
167 context=context,
168 )
169  
170 # Look into the keywords first, maybe it's already there.
171 try:
172 return self.keyword_arguments[name].infer(context)
173 except KeyError:
174 pass
175  
176 # Too many arguments given and no variable arguments.
177 if len(self.positional_arguments) > len(funcnode.args.args):
178 if not funcnode.args.vararg and not funcnode.args.posonlyargs:
179 raise InferenceError(
180 "Too many positional arguments "
181 "passed to {func!r} that does "
182 "not have *args.",
183 call_site=self,
184 func=funcnode,
185 arg=name,
186 context=context,
187 )

190 vararg = self.positional_arguments[len(funcnode.args.args) :]
191 argindex = funcnode.args.find_argname(name)[0]
192 kwonlyargs = {arg.name for arg in funcnode.args.kwonlyargs}
193 kwargs = {
194 key: value
195 for key, value in self.keyword_arguments.items()
196 if key not in kwonlyargs
197 }
198 # If there are too few positionals compared to
199 # what the function expects to receive, check to see
200 # if the missing positional arguments were passed
201 # as keyword arguments and if so, place them into the
202 # positional args list.
203 if len(positional) < len(funcnode.args.args):
204 for func_arg in funcnode.args.args:
205 if func_arg.name in kwargs:
206 arg = kwargs.pop(func_arg.name)
207 positional.append(arg)
208  
209 if argindex is not None:
210 boundnode = getattr(context, "boundnode", None)
211 # 2. first argument of instance/class method
212 if argindex == 0 and funcnode.type in {"method", "classmethod"}:
213 # context.boundnode is None when an instance method is called with
214 # the class, e.g. MyClass.method(obj, ...). In this case, self
215 # is the first argument.
216 if boundnode is None and funcnode.type == "method" and positional:
217 return positional[0].infer(context=context)
218 if boundnode is None:
219 # XXX can do better ?
220 boundnode = funcnode.parent.frame(future=True)
221  
222 if isinstance(boundnode, nodes.ClassDef):
223 # Verify that we're accessing a method
224 # of the metaclass through a class, as in
225 # `cls.metaclass_method`. In this case, the
226 # first argument is always the class.
227 method_scope = funcnode.parent.scope()
228 if method_scope is boundnode.metaclass():
229 return iter((boundnode,))
230  
231 if funcnode.type == "method":
232 if not isinstance(boundnode, Instance):
233 boundnode = boundnode.instantiate_class()
234 return iter((boundnode,))
235 if funcnode.type == "classmethod":
236 return iter((boundnode,))

238 # from the index, so we'll take in account
239 # the extra parameter represented by `self` or `cls`
240 if funcnode.type in {"method", "classmethod"} and boundnode:
241 argindex -= 1
242 # 2. search arg index
243 try:
244 return self.positional_arguments[argindex].infer(context)
245 except IndexError:
246 pass

248 if funcnode.args.kwarg == name:
249 # It wants all the keywords that were passed into
250 # the call site.
251 if self.has_invalid_keywords():
252 raise InferenceError(
253 "Inference failed to find values for all keyword arguments "
254 "to {func!r}: {unpacked_kwargs!r} doesn't correspond to "

256 keyword_arguments=self.keyword_arguments,

259 func=funcnode,

262 )

264 lineno=funcnode.args.lineno,
265 col_offset=funcnode.args.col_offset,
266 parent=funcnode.args,
267 )
268 kwarg.postinit(
269 [(nodes.const_factory(key), value) for key, value in kwargs.items()]

272 if funcnode.args.vararg == name:
273 # It wants all the args that were passed into
274 # the call site.
275 if self.has_invalid_arguments():
276 raise InferenceError(
277 "Inference failed to find values for all positional "
278 "arguments to {func!r}: {unpacked_args!r} doesn't "
279 "correspond to {positional_arguments!r}.",
280 positional_arguments=self.positional_arguments,
281 unpacked_args=self._unpacked_args,
282 call_site=self,
283 func=funcnode,
284 arg=name,
285 context=context,
286 )
287 args = nodes.Tuple(
288 lineno=funcnode.args.lineno,
289 col_offset=funcnode.args.col_offset,
290 parent=funcnode.args,
291 )
292 args.postinit(vararg)
