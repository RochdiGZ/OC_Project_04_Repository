138 def _handle_blacklist(
139 blacklist: Sequence[str], dirnames: list[str], filenames: list[str]
140 ) -> None:
141 """remove files/directories in the black list
142  
143 dirnames/filenames are usually from os.walk
144 """
145 for norecurs in blacklist:
146 if norecurs in dirnames:
147 dirnames.remove(norecurs)
148 elif norecurs in filenames:
149 filenames.remove(norecurs)

157 def _cache_normalize_path(path: str) -> str:
158 """Normalize path with caching."""
159 # _module_file calls abspath on every path in sys.path every time it's
160 # called; on a larger codebase this easily adds up to half a second just
161 # assembling path components. This cache alleviates that.
162 if not path: # don't cache result for ''
163 return _normalize_path(path)
164 return _cache_normalize_path_(path)

245 def _get_relative_base_path(filename: str, path_to_check: str) -> list[str] | None:
246 """Extracts the relative mod path of the file to import from
247  
248 Check if a file is within the passed in path and if so, returns the
249 relative mod path from the one passed in.
250  
251 If the filename is no in path_to_check, returns None
252  
253 Note this function will look for both abs and realpath of the file,
254 this allows to find the relative base path even if the file is a
255 symlink of a file in the passed in path
256  
257 Examples:
258 _get_relative_base_path("/a/b/c/d.py", "/a/b") -> ["c","d"]
259 _get_relative_base_path("/a/b/c/d.py", "/dev") -> None
260 """
261 importable_path = None
262 path_to_check = os.path.normcase(path_to_check)
263 abs_filename = os.path.abspath(filename)
264 if os.path.normcase(abs_filename).startswith(path_to_check):
265 importable_path = abs_filename
266  
267 real_filename = os.path.realpath(filename)
268 if os.path.normcase(real_filename).startswith(path_to_check):
269 importable_path = real_filename
270  
271 # if "var" in path_to_check:
272 # breakpoint()
273  
274 if importable_path:
275 base_path = os.path.splitext(importable_path)[0]
  • E203 Whitespace before ':'
276 relative_base_path = base_path[len(path_to_check) :]
277 return [pkg for pkg in relative_base_path.split(os.sep) if pkg]
278  
279 return None

282 def modpath_from_file_with_callback(
283 filename: str,
284 path: Sequence[str] | None = None,
285 is_package_cb: Callable[[str, list[str]], bool] | None = None,
286 ) -> list[str]:
287 filename = os.path.expanduser(_path_from_filename(filename))
288 paths_to_check = sys.path.copy()
289 if path:
290 paths_to_check += path
291 for pathname in itertools.chain(
292 paths_to_check, map(_cache_normalize_path, paths_to_check)
293 ):
294 if not pathname:
295 continue
296 modpath = _get_relative_base_path(filename, pathname)
297 if not modpath:
298 continue
299 assert is_package_cb is not None
300 if is_package_cb(pathname, modpath[:-1]):
301 return modpath
302  
303 raise ImportError(
304 "Unable to find module for {} in {}".format(filename, ", \n".join(sys.path))
305 )

338 modpath: list[str],

340 context_file: str | None = None,
341 ) -> spec.ModuleSpec:
342 """given a mod path (i.e. split module / package name), return the
343 corresponding file, giving priority to source file over precompiled
344 file if it exists
345  
346 :param modpath:
347 split module's name (i.e name of a module or package split
348 on '.')
349 (this means explicit relative imports that start with dots have
350 empty strings in this list!)
351  
352 :param path:
353 optional list of path where the module or package should be
354 searched (use sys.path if nothing or None is given)
355  
356 :param context_file:
357 context file to consider, necessary if the identifier has been
358 introduced using a relative import unresolvable in the actual
359 context (i.e. modutils)
360  
361 :raise ImportError: if there is no such module in the directory
362  
363 :return:
364 the path to the module's file or None if it's an integrated
365 builtin module such as 'sys'
366 """
367 if context_file is not None:
368 context: str | None = os.path.dirname(context_file)
369 else:
370 context = context_file
371 if modpath[0] == "xml":
372 # handle _xmlplus
373 try:
374 return _spec_from_modpath(["_xmlplus"] + modpath[1:], path, context)
375 except ImportError:
376 return _spec_from_modpath(modpath, path, context)
377 elif modpath == ["os", "path"]:

387 def get_module_part(dotted_name: str, context_file: str | None = None) -> str:
388 """given a dotted name return the module part of the name :
389  
390 >>> get_module_part('astroid.as_string.dump')
391 'astroid.as_string'
392  
393 :param dotted_name: full name of the identifier we are interested in
394  
395 :param context_file:
396 context file to consider, necessary if the identifier has been
397 introduced using a relative import unresolvable in the actual
398 context (i.e. modutils)
399  
400  
401 :raise ImportError: if there is no such module in the directory
402  
403 :return:
404 the module part of the name or None if we have not been able at
405 all to import the given name
406  
407 XXX: deprecated, since it doesn't handle package precedence over module
408 (see #10066)
409 """
410 # os.path trick
411 if dotted_name.startswith("os.path"):
412 return "os.path"
413 parts = dotted_name.split(".")
414 if context_file is not None:
415 # first check for builtin module which won't be considered latter
416 # in that case (path != None)
417 if parts[0] in BUILTIN_MODULES:
418 if len(parts) > 2:
419 raise ImportError(dotted_name)
420 return parts[0]
421 # don't use += or insert, we want a new list to be created !
422 path: list[str] | None = None
423 starti = 0
424 if parts[0] == "":
425 assert (
426 context_file is not None
427 ), "explicit relative import, but no context_file?"
428 path = [] # prevent resolving the import non-relatively
429 starti = 1
430 while parts[starti] == "": # for all further dots: change context
431 starti += 1
432 assert (
433 context_file is not None
434 ), "explicit relative import, but no context_file?"
435 context_file = os.path.dirname(context_file)
436 for i in range(starti, len(parts)):
437 try:
438 file_from_modpath(
  • E203 Whitespace before ':'
439 parts[starti : i + 1], path=path, context_file=context_file
440 )
441 except ImportError:
442 if i < max(1, len(parts) - 2):
443 raise
444 return ".".join(parts[:i])
445 return dotted_name

448 def get_module_files(
449 src_directory: str, blacklist: Sequence[str], list_all: bool = False
450 ) -> list[str]:
451 """given a package directory return a list of all available python
452 module's files in the package and its subpackages
453  
454 :param src_directory:
455 path of the directory corresponding to the package
456  
457 :param blacklist: iterable
458 list of files or directories to ignore.
459  
460 :param list_all:
461 get files from all paths, including ones without __init__.py
462  
463 :return:
464 the list of all available python module's files in the package and
465 its subpackages
466 """
467 files: list[str] = []
468 for directory, dirnames, filenames in os.walk(src_directory):
469 if directory in blacklist:
470 continue
471 _handle_blacklist(blacklist, dirnames, filenames)
472 # check for __init__.py
473 if not list_all and "__init__.py" not in filenames:
474 dirnames[:] = ()
475 continue
476 for filename in filenames:
477 if _is_python_file(filename):
478 src = os.path.join(directory, filename)
479 files.append(src)
480 return files

483 def get_source_file(filename: str, include_no_ext: bool = False) -> str:
484 """given a python module's file name return the matching source file
485 name (the filename will be returned identically if it's already an
486 absolute path to a python source file...)
487  
488 :param filename: python module's file name
489  
490 :raise NoSourceFile: if no source file exists on the file system
491  
492 :return: the absolute path of the source file if it exists
493 """
494 filename = os.path.abspath(_path_from_filename(filename))
495 base, orig_ext = os.path.splitext(filename)
496 for ext in PY_SOURCE_EXTS:
497 source_path = f"{base}.{ext}"
498 if os.path.exists(source_path):
499 return source_path
500 if include_no_ext and not orig_ext and os.path.exists(base):
501 return base
502 raise NoSourceFile(filename)

514 def is_standard_module(modname: str, std_path: Iterable[str] | None = None) -> bool:
515 """try to guess if a module is a standard python module (by default,
516 see `std_path` parameter's description)
517  
518 :param modname: name of the module we are interested in
519  
520 :param std_path: list of path considered has standard
521  
522 :return:
523 true if the module:
524 - is located on the path listed in one of the directory in `std_path`
525 - is a built-in module
526 """
527 modname = modname.split(".")[0]
528 try:
529 filename = file_from_modpath([modname])
530 except ImportError:
531 # import failed, i'm probably not so wrong by supposing it's
532 # not standard...
533 return False
534 # modules which are not living in a file are considered standard
535 # (sys and __builtin__ for instance)
536 if filename is None:
537 # we assume there are no namespaces in stdlib
538 return not util.is_namespace(modname)
539 filename = _normalize_path(filename)
540 for path in EXT_LIB_DIRS:
541 if filename.startswith(_cache_normalize_path(path)):
542 return False
543 if std_path is None:
544 std_path = STD_LIB_DIRS
545  
546 return any(filename.startswith(_cache_normalize_path(path)) for path in std_path)

549 def is_relative(modname: str, from_file: str) -> bool:
550 """return true if the given module name is relative to the given
551 file name
552  
553 :param modname: name of the module we are interested in
554  
555 :param from_file:
556 path of the module from which modname has been imported
557  
558 :return:
559 true if the module has been imported relatively to `from_file`
560 """
561 if not os.path.isdir(from_file):
562 from_file = os.path.dirname(from_file)
563 if from_file in sys.path:
564 return False
565 return bool(
566 importlib.machinery.PathFinder.find_spec(
567 modname.split(".", maxsplit=1)[0], [from_file]
568 )
569 )

575 def _spec_from_modpath(
576 modpath: list[str],
577 path: Sequence[str] | None = None,
578 context: str | None = None,
579 ) -> spec.ModuleSpec:
580 """given a mod path (i.e. split module / package name), return the
581 corresponding spec
582  
583 this function is used internally, see `file_from_modpath`'s
584 documentation for more information
585 """
586 assert modpath
587 location = None
588 if context is not None:
589 try:
590 found_spec = spec.find_spec(modpath, [context])
591 location = found_spec.location
592 except ImportError:
593 found_spec = spec.find_spec(modpath, path)
594 location = found_spec.location
595 else:
596 found_spec = spec.find_spec(modpath, path)
597 if found_spec.type == spec.ModuleType.PY_COMPILED:
598 try:
599 assert found_spec.location is not None
600 location = get_source_file(found_spec.location)
601 return found_spec._replace(
602 location=location, type=spec.ModuleType.PY_SOURCE
603 )
604 except NoSourceFile:
605 return found_spec._replace(location=location)
606 elif found_spec.type == spec.ModuleType.C_BUILTIN:
607 # integrated builtin module
608 return found_spec._replace(location=None)
609 elif found_spec.type == spec.ModuleType.PKG_DIRECTORY:
610 assert found_spec.location is not None
611 location = _has_init(found_spec.location)
612 return found_spec._replace(location=location, type=spec.ModuleType.PY_SOURCE)
613 return found_spec

624 def _has_init(directory: str) -> str | None:
625 """if the given directory has a valid __init__ file, return its path,
626 else return None
627 """
628 mod_or_pack = os.path.join(directory, "__init__")
629 for ext in PY_SOURCE_EXTS + ("pyc", "pyo"):
630 if os.path.exists(mod_or_pack + "." + ext):
