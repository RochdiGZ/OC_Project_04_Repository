390 def optik_option(
391 self, provider: ConfigProvider, opt: str, optdict: OptionDict
392 ) -> tuple[list[str], OptionDict]: # pragma: no cover
393 """DEPRECATED: Get our personal option definition and return a suitable form for
394 use with optik/optparse.
395 """
396 warnings.warn(
397 "optik_option has been deprecated. Parsing of option dictionaries should be done "
398 "automatically by initializing an ArgumentsProvider.",
399 DeprecationWarning,
400 )
401 optdict = copy.copy(optdict)
402 if "action" in optdict:
403 self._nocallback_options[provider] = opt
404 else:
405 optdict["action"] = "callback"
406 optdict["callback"] = self.cb_set_provider_option
407 # default is handled here and *must not* be given to optik if you
408 # want the whole machinery to work
409 if "default" in optdict:
410 if (
411 "help" in optdict
412 and optdict.get("default") is not None
413 and optdict["action"] not in ("store_true", "store_false")
414 ):
415 optdict["help"] += " [current: %default]" # type: ignore[operator]
416 del optdict["default"]
417 args = ["--" + str(opt)]
418 if "short" in optdict:
419 self._short_options[optdict["short"]] = opt # type: ignore[index]
420 args.append("-" + optdict["short"]) # type: ignore[operator]
421 del optdict["short"]
422 # cleanup option definition dict before giving it to optik
423 for key in list(optdict.keys()):
424 if key not in self._optik_option_attrs:
425 optdict.pop(key)
426 return args, optdict

716 # Add help comment
717 help_msg = optdict.get("help", "")
718 assert isinstance(help_msg, str)
719 help_text = textwrap.wrap(help_msg, width=79)
720 for line in help_text:
721 group_table.add(tomlkit.comment(line))
722  
723 # Get current value of option
724 value = getattr(self.config, optname.replace("-", "_"))
725  
726 # Create a comment if the option has no value
727 if not value:
728 group_table.add(tomlkit.comment(f"{optname} ="))
729 group_table.add(tomlkit.nl())
730 continue
731  
732 # Skip deprecated options
733 if "kwargs" in optdict:
734 assert isinstance(optdict["kwargs"], dict)
735 if "new_names" in optdict["kwargs"]:
736 continue
737  
738 # Tomlkit doesn't support regular expressions
739 if isinstance(value, re.Pattern):
740 value = value.pattern
741 elif isinstance(value, (list, tuple)) and isinstance(
742 value[0], re.Pattern
743 ):
744 value = [i.pattern for i in value]
745  
746 # Handle tuples that should be strings
747 if optdict.get("type") == "py_version":
748 value = ".".join(str(i) for i in value)
749  
750 # Add to table
751 group_table.add(optname, value)
752 group_table.add(tomlkit.nl())
753  
754 assert group.title
755 pylint_tool_table.add(group.title.lower(), group_table)
756  
757 toml_string = tomlkit.dumps(toml_doc)
758  
759 # Make sure the string we produce is valid toml and can be parsed
760 tomllib.loads(toml_string)
761  
762 return toml_string
