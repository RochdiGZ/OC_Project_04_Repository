56 # Because parsing LilyPond input is very tricky (and in fact
57 # impossible without executing LilyPond when there is Scheme
58 # code in the file), this lexer does not try to recognize
59 # lexical modes. Instead, it catches the most frequent pieces
60 # of syntax, and, above all, knows about many kinds of builtins.
61  
62 # In order to parse embedded Scheme, this lexer subclasses the SchemeLexer.
63 # It redefines the 'root' state entirely, and adds a rule for #{ #}
64 # to the 'value' state. The latter is used to parse a Scheme expression
65 # after #.
66  
67 def get_tokens_unprocessed(self, text):
68 """Highlight Scheme variables as LilyPond builtins when applicable."""
69 for index, token, value in super().get_tokens_unprocessed(text):
70 if token is Token.Name.Function or token is Token.Name.Variable:
71 if value in scheme_functions:
72 token = Token.Name.Builtin.SchemeFunction
73 elif token is Token.Name.Builtin:
74 token = Token.Name.Builtin.SchemeBuiltin
75 yield index, token, value

77 tokens = {
78 "root": [
79 # Whitespace.
80 (r"\s+", Token.Text.Whitespace),
81  
82 # Multi-line comment. These are non-nestable.
83 (r"%\{.*?%\}", Token.Comment.Multiline),
84  
85 # Simple comment.
86 (r"%.*?$", Token.Comment.Single),
87  
88 # End of embedded LilyPond in Scheme.
89 (r"#\}", Token.Punctuation, "#pop"),
90  
91 # Embedded Scheme, starting with # ("delayed"),
92 # or $ (immediate). #@ and and $@ are the lesser known
93 # "list splicing operators".
94 (r"[#$]@?", Token.Punctuation, "value"),
95  
96 # Any kind of punctuation:
97 # - sequential music: { },
98 # - parallel music: << >>,
99 # - voice separator: << \\ >>,
100 # - chord: < >,
101 # - bar check: |,
102 # - dot in nested properties: \revert NoteHead.color,
103 # - equals sign in assignments and lists for various commands:
104 # \override Stem.color = red,
105 # - comma as alternative syntax for lists: \time 3,3,2 4/4,
106 # - colon in tremolos: c:32,
107 # - double hyphen in lyrics: li -- ly -- pond,
108 (r"\\\\|--|[{}<>=.,:|]", Token.Punctuation),
109  
110 # Pitch, with optional octavation marks, octave check,
111 # and forced or cautionary accidental.
112 (words(pitches, suffix=r"=?[',]*!?\??" + NAME_END_RE), Token.Pitch),
113  
114 # String, optionally with direction specifier.
115 (r'[\-_^]?"', Token.String, "string"),
116  
117 # Numbers.
  • E261 At least two spaces before inline comment
118 (r"-?\d+\.\d+", Token.Number.Float), # 5. and .5 are not allowed
119 (r"-?\d+/\d+", Token.Number.Fraction),
120 # Integer, or duration with optional augmentation dots. We have no
121 # way to distinguish these, so we highlight them all as numbers.
122 (r"-?(\d+|\\longa|\\breve)\.*", Token.Number),
123 # Separates duration and duration multiplier highlighted as fraction.
124 (r"\*", Token.Number),
125  
126 # Ties, slurs, manual beams.
127 (r"[~()[\]]", Token.Name.Builtin.Articulation),
128  
129 # Predefined articulation shortcuts. A direction specifier is
130 # required here.
131 (r"[\-_^][>^_!.\-+]", Token.Name.Builtin.Articulation),
132  
133 # Fingering numbers, string numbers.
134 (r"[\-_^]?\\?\d+", Token.Name.Builtin.Articulation),
135

138 (builtin_words(pitch_language_names, "disallowed"), Token.Name.PitchLanguage),
139 (builtin_words(clefs, "disallowed"), Token.Name.Builtin.Clef),
140 (builtin_words(scales, "mandatory"), Token.Name.Builtin.Scale),
141 (builtin_words(repeat_types, "disallowed"), Token.Name.Builtin.RepeatType),
142 (builtin_words(units, "mandatory"), Token.Number),
143 (builtin_words(chord_modifiers, "disallowed"), Token.ChordModifier),
144 (builtin_words(music_functions, "mandatory"), Token.Name.Builtin.MusicFunction),
145 (builtin_words(dynamics, "mandatory"), Token.Name.Builtin.Dynamic),
146 # Those like slurs that don't take a backslash are covered above.
147 (builtin_words(articulations, "mandatory"), Token.Name.Builtin.Articulation),
148 (builtin_words(music_commands, "mandatory"), Token.Name.Builtin.MusicCommand),
149 (builtin_words(markup_commands, "mandatory"), Token.Name.Builtin.MarkupCommand),
150 (builtin_words(grobs, "disallowed"), Token.Name.Builtin.Grob),
151 (builtin_words(translators, "disallowed"), Token.Name.Builtin.Translator),
152 # Optional backslash because of \layout { \context { \Score ... } }.
153 (builtin_words(contexts, "optional"), Token.Name.Builtin.Context),
154 (builtin_words(context_properties, "disallowed"), Token.Name.Builtin.ContextProperty),
155 (builtin_words(grob_properties, "disallowed"),
156 Token.Name.Builtin.GrobProperty,
157 "maybe-subproperties"),
158 # Optional backslashes here because output definitions are wrappers

161 (builtin_words(paper_variables, "optional"), Token.Name.Builtin.PaperVariable),
162 (builtin_words(header_variables, "optional"), Token.Name.Builtin.HeaderVariable),
163  
164 # Other backslashed-escaped names (like dereferencing a
165 # music variable), possibly with a direction specifier.
166 (r"[\-_^]?\\.+?" + NAME_END_RE, Token.Name.BackslashReference),
167  
168 # Definition of a variable. Support assignments to alist keys
169 # (myAlist.my-key.my-nested-key = \markup \spam \eggs).
170 (r"([^\W\d]|-)+(?=([^\W\d]|[\-.])*\s*=)", Token.Name.Lvalue),
171  
172 # Virtually everything can appear in markup mode, so we highlight
173 # as text. Try to get a complete word, or we might wrongly lex
174 # a suffix that happens to be a builtin as a builtin (e.g., "myStaff").
175 (r"([^\W\d]|-)+?" + NAME_END_RE, Token.Text),
176 (r".", Token.Text),
177 ],
178 "string": [
179 (r'"', Token.String, "#pop"),
180 (r'\\.', Token.String.Escape),
181 (r'[^\\"]+', Token.String),
182 ],
183 "value": [

188 ],

198 ]
