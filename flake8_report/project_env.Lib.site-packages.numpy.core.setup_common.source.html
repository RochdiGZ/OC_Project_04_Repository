381 def long_double_representation(lines):
382 """Given a binary dump as given by GNU od -b, look for long double
383 representation."""
384  
385 # Read contains a list of 32 items, each item is a byte (in octal
386 # representation, as a string). We 'slide' over the output until read is of
387 # the form before_seq + content + after_sequence, where content is the long double
388 # representation:
389 # - content is 12 bytes: 80 bits Intel representation
390 # - content is 16 bytes: 80 bits Intel representation (64 bits) or quad precision
391 # - content is 8 bytes: same as double (not implemented yet)
392 read = [''] * 32
393 saw = None
394 for line in lines:
395 # we skip the first word, as od -b output an index at the beginning of
396 # each line
397 for w in line.split()[1:]:
398 read.pop(0)
399 read.append(w)
400  
401 # If the end of read is equal to the after_sequence, read contains
402 # the long double
403 if read[-8:] == _AFTER_SEQ:
404 saw = copy.copy(read)
405 # if the content was 12 bytes, we only have 32 - 8 - 12 = 12
406 # "before" bytes. In other words the first 4 "before" bytes went
407 # past the sliding window.
408 if read[:12] == _BEFORE_SEQ[4:]:
409 if read[12:-8] == _INTEL_EXTENDED_12B:
410 return 'INTEL_EXTENDED_12_BYTES_LE'
411 if read[12:-8] == _MOTOROLA_EXTENDED_12B:
412 return 'MOTOROLA_EXTENDED_12_BYTES_BE'
413 # if the content was 16 bytes, we are left with 32-8-16 = 16
414 # "before" bytes, so 8 went past the sliding window.
415 elif read[:8] == _BEFORE_SEQ[8:]:
416 if read[8:-8] == _INTEL_EXTENDED_16B:
417 return 'INTEL_EXTENDED_16_BYTES_LE'
418 elif read[8:-8] == _IEEE_QUAD_PREC_BE:
419 return 'IEEE_QUAD_BE'
420 elif read[8:-8] == _IEEE_QUAD_PREC_LE:
421 return 'IEEE_QUAD_LE'
422 elif read[8:-8] == _IBM_DOUBLE_DOUBLE_LE:
423 return 'IBM_DOUBLE_DOUBLE_LE'
424 elif read[8:-8] == _IBM_DOUBLE_DOUBLE_BE:
425 return 'IBM_DOUBLE_DOUBLE_BE'
426 # if the content was 8 bytes, left with 32-8-8 = 16 bytes
427 elif read[:16] == _BEFORE_SEQ:
428 if read[16:-8] == _IEEE_DOUBLE_LE:
429 return 'IEEE_DOUBLE_LE'
430 elif read[16:-8] == _IEEE_DOUBLE_BE:
431 return 'IEEE_DOUBLE_BE'
432  
433 if saw is not None:
434 raise ValueError("Unrecognized format (%s)" % saw)
435 else:
436 # We never detected the after_sequence
437 raise ValueError("Could not lock sequences (%s)" % saw)
