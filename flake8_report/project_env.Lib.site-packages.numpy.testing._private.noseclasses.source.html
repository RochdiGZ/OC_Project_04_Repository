30 def _from_module(self, module, object):
31 """
32 Return true if the given object is defined in the given
33 module.
34 """
35 if module is None:
36 return True
37 elif inspect.isfunction(object):
38 return module.__dict__ is object.__globals__
39 elif inspect.isbuiltin(object):
40 return module.__name__ == object.__module__
41 elif inspect.isclass(object):
42 return module.__name__ == object.__module__
43 elif inspect.ismethod(object):
44 # This one may be a bug in cython that fails to correctly set the
45 # __module__ attribute of methods, but since the same error is easy
46 # to make by extension code writers, having this safety in place
47 # isn't such a bad idea
48 return module.__name__ == object.__self__.__class__.__module__
49 elif inspect.getmodule(object) is not None:
50 return module is inspect.getmodule(object)
51 elif hasattr(object, '__module__'):
52 return module.__name__ == object.__module__
53 elif isinstance(object, property):
54 return True # [XX] no way not be sure.
55 else:
56 raise ValueError("object must be a class or function")
